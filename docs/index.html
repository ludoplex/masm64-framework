Clean, Readable Assembly

The framework's macros make x64 assembly approachable

main.asm 
; Simple MASM64 Framework Example
INCLUDE core\abi64.inc
INCLUDE core\stack64.inc
INCLUDE lib\arena64\arena64.inc

.DATA
szHello    DB "Hello from MASM64!", 0

.CODE
main PROC FRAME
    ; Prologue with proper stack alignment
    PROLOGUE 0, 64
    
    ; Create arena allocator (1MB)
    mov rcx, 1024 * 1024
    mov rdx, 64 * 1024
    mov r8d, ARENA_GROW_COMMIT
    call Arena_Create
    ASSERT_RAX                  ; Verify allocation
    
    ; Print message
    PRINT_STRING szHello
    
    ; Epilogue and exit
    EPILOGUE
    EXIT_PROCESS 0
main ENDP
END
